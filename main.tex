\section{QUESTION No-2.28(Vectors)}
Find the ratio in  which the line segment joining the points \myvec{4 \\ 8 \\ 10} and \myvec{6 \\ 10 \\ -8} is divided by YZ-plane. 
\section{Solution}
\begin{enumerate}
\item Given 
\begin{align}
  \vec{A} = \myvec{4 \\ 8 \\ 10}, \vec{B}=\myvec{6 \\ 10\\ -8} \label{eq:1}
\end{align}
Let the corresponding points on the $YZ$ axis be \myvec{0\\x\\y}
\\
If the ratio be $k:1$
\\
using, \\
\begin{align}
\vec{C}=\frac{k\myvec{B}+\myvec{A}}{(k+1)}\\
\vec{k}=\frac{A-C}{C-B}\\
\implies\myvec{C-B}\myvec{k}=\vec{A-C}\\
\end{align}
Let
\begin{align}
\vec{C}=\myvec{0\\y\\z}\\
\myvec{C-B}\myvec{k}=\myvec{A-C}\\
\myvec{-6\\y-10\\z+8}(k) =\myvec{4\\8-y\\10-z}
\end{align}
 As this is divided YZ plane, x-coordinate will zero. \\
 \begin{align}
  \vec{e}=\myvec{1\\0\\0}\\
  \myvec{1 & 0& 0}\myvec{-6\\y-10\\z+8}\myvec{k}=\myvec{1 & 0 & 0}\myvec{4\\8-y\\10-z}
  \\
  \implies
  \myvec{-6}\myvec{k}=\myvec{4}\\
  \vec{k}=\frac{-2}{3}
 \end{align}
 So, required ratio is 2:3 and line segment is divided externally.
\end{enumerate}
